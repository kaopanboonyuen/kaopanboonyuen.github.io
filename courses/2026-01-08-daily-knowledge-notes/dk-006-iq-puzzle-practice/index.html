    solutions = []
    def backtrack(row, queens, xy_diff, xy_sum):
        if row == n:
            solutions.append(queens)
            return
        for col in range(n):
            if col not in queens and row-col not in xy_diff and row+col not in xy_sum:
                backtrack(row+1, queens+[col], xy_diff+[row-col], xy_sum+[row+col])
    backtrack(0,[],[],[])
