üî¥ Problem 15 ‚Äî Smallest Window with 2 ERR

logs = ["OK","ERR","OK","ERR","ERR","OK"]
‚úÖ Solution ‚Äî Sliding Window 
left = err = 0
best = float("inf")

for right in range(len(logs)):
    if logs[right] == "ERR":
        err += 1

    while err >= 2:
        best = min(best, right - left + 1)
        if logs[left] == "ERR":
            err -= 1
        left += 1

print(best)

üî¥ Problem 17 ‚Äî Cycle Detection (Graph)

deps = {"A":["B"],"B":["C"],"C":[]}
‚úÖ Solution ‚Äî DFS 
vis, path = set(), set()

def dfs(n):
    if n in path: return True
    if n in vis: return False
    path.add(n)
    for nei in deps[n]:
        if dfs(nei): return True
    path.remove(n)
    vis.add(n)
    return False

print(any(dfs(x) for x in deps))
